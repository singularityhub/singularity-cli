[docs]def parse_verbosity(self, args):
    """parse_verbosity will take an argument object, and return the args
    passed (from a dictionary) to a list

    Parameters
    ==========
    args: the argparse argument objects

    """

    flags = []

    if args.silent:
        flags.append("--silent")
    elif args.quiet:
        flags.append("--quiet")
    elif args.debug:
        flags.append("--debug")
    elif args.verbose:
        flags.append("-" + "v" * args.verbose)

    return flags
